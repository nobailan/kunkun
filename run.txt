# ============================================================
# Kunkun 命令速查
# ============================================================

# ── 环境 ──
conda activate graph


# ============================================================
# 一、启动模式
# ============================================================

# 交互模式（进入 REPL 循环）
kunkun-interactive
# 或
kunkun

# 交互模式带详细输出（显示 Thinking + 状态）
kunkun -v

# 单次执行
kunkun "你的任务描述"

# 单次执行 + 详细输出
kunkun -v "你的任务描述"

# 隐藏 Thinking 块
kunkun --hide-thinking "你的任务描述"

# 指定模型
kunkun -m deepseek-v4-pro "你的任务描述"

# 指定工作目录
kunkun -w E:\other-project "你的任务描述"

# 指定最大轮次
kunkun --max-turns 30 "你的任务描述"

# 生成 HTML 评测仪表盘（自动打开浏览器）
kunkun --dashboard

# FlowForge 模式（JSON 输出）
kunkun --flowforge "你的任务描述"

# Cron 调度器模式（后台运行，Ctrl+C 停止）
kunkun --cron

# 查看版本
kunkun --version

# ============================================================
# 四、交互模式内可用命令
# ============================================================

# 退出交互模式
#   输入 exit 或 quit 或 q 或 Ctrl+C

# 工具调用（Agent 自动使用，用户无需手动调用）
#   bash, read_file, write_file, glob, grep, edit
#   remember, recall, skill_load
#   websearch, webfetch
#   agent, todowrite, grpo
#   findsymbol, gotodef, findrefs, session_search

# 记忆操作
#   remember target=memory name=xxx content=xxx  → 保存事实
#   remember target=skill name=xxx content=xxx   → 保存约定/规范
#   recall query=xxx                              → 搜索记忆


# ============================================================
# 五、配置（.env 文件）
# ============================================================

# KUN_API_KEY=sk-...           # DeepSeek API 密钥（必填）
# KUN_MODEL=deepseek-v4-pro    # 主模型
# KUN_LIGHT_MODEL=deepseek-v4-flash  # 评测/反思用轻模型
# KUN_BASE_URL=https://api.deepseek.com
# KUN_WORKSPACE=.              # 工作目录
# KUN_MAX_TURNS=50             # 最大轮次
# KUN_MAX_TOKENS_PER_TURN=64000
# KUN_PERMISSION_MODE=accept_edits  # default | accept_edits | bypass
